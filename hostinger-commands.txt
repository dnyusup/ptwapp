# Perintah yang harus dijalankan di terminal Hostinger setelah upload

# 1. Clear semua cache
php artisan route:clear
php artisan config:clear  
php artisan view:clear
php artisan cache:clear

# 2. Update autoloader
composer dump-autoload

# 3. Jalankan migration (jika ada yang baru)
php artisan migrate

# 4. Cache ulang untuk performance
php artisan route:cache
php artisan config:cache
php artisan view:cache

# 5. Cek apakah routes sudah terdaftar
php artisan route:list --name=loto-isolations
